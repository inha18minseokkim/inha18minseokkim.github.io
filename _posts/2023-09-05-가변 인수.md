이런식으로 나옴
한 메서드에 가변인수는 하나, 맨 끝에만 들어갈 수 있다.
만약 빌더패턴을 사용한다면 한 함수에 가변인수를 각각 하나씩 넣어 가변인수를 여러 개 쓸 수 있는 효과를 낼 수 있다고 함.

```java
class Muyaho {
    public List<Integer> a;
    public List<Integer> b;
    public List<Integer> c;
    public Muyaho(){}
    private Muyaho(MuyahoBuilder builder){
        this.a = builder.a;
        this.b = builder.b;
        this.c = builder.c;
    }

    public static class MuyahoBuilder {
        private List<Integer> a;
        private List<Integer> b;
        private List<Integer> c;
        public MuyahoBuilder a(int... a){
            this.a = Arrays.stream(a).boxed().toList();
            return this;
        }
        public MuyahoBuilder b(int... b){
            this.b = Arrays.stream(b).boxed().toList();
            return this;
        }
        public MuyahoBuilder c(int... c){
            this.c = Arrays.stream(c).boxed().toList();
            return this;
        }
        public Muyaho build(){
            return new Muyaho(this);
        }

    }

}
@SpringBootApplication
public class TestApplication {
    private final static Logger log = LoggerFactory.getLogger(TestApplication.class);

    public static void main(String[] args)  {
        Muyaho muyaho = new Muyaho.MuyahoBuilder()
                .a(1,2,3)
                .b(4,5,6)
                .c(7,8,9)
                .build();
        System.out.println(muyaho.a);
        System.out.println(muyaho.b);
        System.out.println(muyaho.c);
    }
}
```
